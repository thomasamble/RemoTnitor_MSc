  mutate(Species = factor(Species, levels = zymobiomics_species),
         `Expected relative abundance` = case_when(Species %in% c("Pseudomonas aeruginosa",
                                                                  "Escherichia coli",
                                                                  "Salmonella enterica",
                                                                  "Limosilactobacillus fermentum",
                                                                  "Enterococcus faecalis",
                                                                  "Staphylococcus aureus",
                                                                  "Listeria monocytogenes",
                                                                  "Bacillus subtilis") ~ 12 / 100,
                                                   Species %in% c("Saccharomyces cerevisiae",
                                                                  "Cryptococcus neoformans") ~ 2 / 100),
         Difference = `Observed relative abundance` - `Expected relative abundance`,
         log2FC = log2(`Observed relative abundance` / `Expected relative abundance`)) %>% 
  select(Species, `Observed relative abundance`, `Expected relative abundance`, Difference, log2FC) %>% 
  arrange(Species)

5 Statistics

Aggregating the rarefied phyloseq object to phylum rank, and transforming to relative abundances (“compositional”). Then filtering to only keep bacterial taxa, and filtering out phyla with a mean relative abundance below 1% across all samples. Then fitting an ANOVA with sample-level phylum relative abundance as response variable, and an interaction term with season * location * phylum (main effects and interactions) as explanatory variables. Then performing a TukeyHSD-test on the ANOVA, and extracting contrasts where the compared groups belong to the same phylum, and have p-values below 0.05.

phylum_data <- rarefied_ps %>%
  tax_agg("phylum") %>%
  tax_transform("compositional") %>% 
  psmelt() %>% 
  #filter(Sample != "WKR24-79") %>% 
  filter(top_rank == "Bacteria") %>% 
  group_by(phylum) %>%
  mutate(mean_abundance = mean(Abundance)) %>% 
  ungroup() %>%
  filter(mean_abundance > 0.01)

# ANOVA
## Note: Relative abundance is not independent, increase one phylums abundance -> decrease others (violates assumptions)
a <- aov(Abundance ~ season * location * phylum,
    data = phylum_data)

# Tukey HSD
Tukey <- TukeyHSD(a, conf.level = 0.95)

same_phylum <- Tukey$`season:location:phylum` %>%
  as.data.frame() %>% 
  rownames_to_column() %>%
  separate_wider_delim(rowname,
                       delim = "-",
                       names = c("obs_1", "obs_2")) %>% 
  separate_wider_delim(obs_1,
                       delim = ":",
                       names = c("season_1", "location_1", "phylum_1")) %>% 
  separate_wider_delim(obs_2,
                       delim = ":",
                       names = c("season_2", "location_2", "phylum_2")) %>% 
  filter(phylum_1 == phylum_2,
         `p adj` < 0.05)
